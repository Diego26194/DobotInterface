# /etc/systemd/system/inicializacion_encendido.service
[Unit]
Description=Inicialización del robot colaborativo (encendido)
After=roscore.service

[Service]
Type=simple
User=dobot
Environment=ROS_DISTRO=noetic
Environment=DISPLAY=:0  # Ajusta si tu entorno gráfico está en otra pantalla
Environment=XAUTHORITY=/home/dobot/.Xauthority  # Asegúrate de que esta ruta sea correcta
ExecStart=/bin/bash -c 'source /opt/ros/noetic/setup.bash; source /home/dobot/catkin_ws/devel/setup.bash; exec /opt/ros/noetic/bin/roslaunch cobot_controladores inicializacion_encendido.launch'

Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
